import java.util.Scanner;

public class VernamCipher {

    // Encryption
    public static String encrypt(String text, String key) {
        StringBuilder result = new StringBuilder();
        text = text.toUpperCase();
        key = key.toUpperCase();

        int j = 0;
        for (int i = 0; i < text.length(); i++) {
            char c = text.charAt(i);
            if (Character.isLetter(c)) {
                char k = key.charAt(j % key.length());
                result.append((char) (((c - 'A') + (k - 'A')) % 26 + 'A'));
                j++;
            } else {
                result.append(c);
            }
        }
        return result.toString();
    }

    // Decryption
    public static String decrypt(String text, String key) {
        StringBuilder result = new StringBuilder();
        text = text.toUpperCase();
        key = key.toUpperCase();

        int j = 0;
        for (int i = 0; i < text.length(); i++) {
            char c = text.charAt(i);
            if (Character.isLetter(c)) {
                char k = key.charAt(j % key.length());
                result.append((char) (((c - 'A') - (k - 'A') + 26) % 26 + 'A'));
                j++;
            } else {
                result.append(c);
            }
        }
        return result.toString();
    }

    // Main
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("NOTE: While entering Key for Vernam cipher please enter a string of any length (letters only)");

        System.out.print("Enter text to encrypt: ");
        String text = sc.nextLine();

        // Text Validation
        while (text.trim().isEmpty()) {
            System.out.print(" Text cannot be empty. Please enter valid text: ");
            text = sc.nextLine();
        }

        String key = "";
        boolean validKey = false;

        //  Key Validation
        while (!validKey) {
            System.out.print("Enter Key: ");
            key = sc.next();

            if (key.trim().isEmpty()) {
                System.out.println(" Key cannot be empty!");
                continue;
            }

            if (!key.matches("[a-zA-Z]+")) {
                System.out.println(" Invalid key! Only letters A–Z or a–z are allowed. Please re-enter.");
                continue;
            }

            if (key.length() < text.replaceAll("[^a-zA-Z]", "").length()) {
                System.out.println(" Key must be at least as long as the alphabetic portion of the text for Vernam Cipher.");
                continue;
            }

            validKey = true;
        }

        System.out.println("Original text: " + text);
        String encrypted = encrypt(text, key);
        System.out.println("Encrypted text: " + encrypted);

        String decrypted = decrypt(encrypted, key);
        System.out.println("Decrypted text: " + decrypted);

        sc.close();
    }
}
