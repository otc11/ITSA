import java.util.Scanner;

public class VernamCipher {

    public static String vernman(String text, String key) {
        text = text.toUpperCase();
        key = key.toUpperCase();
        StringBuilder result = new StringBuilder();

        for (int i = 0; i < text.length(); i++) {
            char p = text.charAt(i);
            char k = key.charAt(i);
            char value = (char) (((p - 'A') ^ (k - 'A')) + 'A');
            result.append(value);
        }

        return result.toString();
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter the plaintext: ");
        String text = scanner.nextLine();

        System.out.print("Enter the key: ");
        String key = scanner.nextLine();

        if (text.length() != key.length()) {
            System.out.println("The length of the text and key should be the same.");
            scanner.close();
            return;
        }

        String cipher = vernman(text, key);
        String plain = vernman(cipher, key);

        System.out.println("Ciphertext: " + cipher);
        System.out.println("Decrypted: " + plain);

        scanner.close();
    }
}
