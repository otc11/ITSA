import java.util.Scanner;

public class MonoAlphabeticCipher {

    public static String encrypt(String text, String key) {
        StringBuilder ans = new StringBuilder();
        text = text.toUpperCase();
        key = key.toUpperCase();

        for (char ch : text.toCharArray()) {
            if (Character.isLetter(ch)) {
                int index = ch - 'A';
                ans.append(key.charAt(index));
            } else {
                ans.append(ch);
            }
        }
        return ans.toString();
    }

    public static String decrypt(String text, String key) {
        StringBuilder ans = new StringBuilder();
        text = text.toUpperCase();
        key = key.toUpperCase();

        for (char ch : text.toCharArray()) {
            if (Character.isLetter(ch)) {
                int index = key.indexOf(ch);
                ans.append((char) (index + 'A'));
            } else {
                ans.append(ch);
            }
        }
        return ans.toString();
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("NOTE: While entering Key for Monoalphabetic cipher please enter 26 character string in any order");

        System.out.print("Enter text to encrypt: ");
        String text = sc.nextLine();

        String key;
        while (true) {
            System.out.print("Enter Key (exactly 26 letters): ");
            key = sc.next();

            //  Check if all characters are alphabetic
            if (!key.matches("[a-zA-Z]+")) {
                System.out.println(" Alphanumeric characters not allowed! Please use only letters Aâ€“Z.");
                continue;
            }

            //  Check if key has exactly 26 letters
            if (key.length() != 26) {
                System.out.println(" Invalid key length! The key must contain exactly 26 letters.");
                continue;
            }

            break; //  Valid key
        }

        System.out.println("Original text: " + text);
        String encrypted = encrypt(text, key);
        System.out.println("Encrypted text: " + encrypted);

        String decrypted = decrypt(encrypted, key);
        System.out.println("Decrypted text: " + decrypted);

        sc.close();
    }
}
